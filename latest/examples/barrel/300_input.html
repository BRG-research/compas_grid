 18def from_barrel_vault(
 19    span: float = 6.0,
 20    length: float = 6.0,
 21    thickness: float = 0.25,
 22    rise: float = 0.6,
 23    vou_span: int = 9,
 24    vou_length: int = 6,
 25    zero_is_centerline_or_lowestpoint: bool = False,
 26) -> tuple[list[Mesh], list[Frame]]:
 27    """
 28    Creates block elements from the barrel vault geometry.
 29
 30    Parameters
 31    ----------
 32    span : float
 33        span of the vault
 34    length : float
 35        length of the vault perpendicular to the span
 36    thickness : float
 37        thickness of the vault
 38    rise : float
 39        rise of the vault from 0.0 to middle axis of the vault thickness
 40    vou_span : int
 41        number of voussoirs in the span direction
 42    vou_length : int
 43        number of voussoirs in the length direction
 44    zero_is_centerline_or_lowestpoint : bool
 45        if True, the lowest point of the vault is at the center line of the arch, otherwise the center line of the vault is lowest mesh z-coordinate.
 46
 47    Returns
 48    -------
 49    tuple[list[`compas.datastructures.Mesh`], list[`compas.geometry.Frame`]]
 50        A tuple containing the block meshes and the frames of the blocks.
 51    """
 52    radius = rise / 2 + span**2 / (8 * rise)
 53    top = [0, 0, rise]
 54    left = [-span / 2, 0, 0]
 55    center = [0.0, 0.0, rise - radius]
 56    vector = subtract_vectors(left, center)
 57    springing = angle_vectors(vector, [-1.0, 0.0, 0.0])
 58    sector = radians(180) - 2 * springing
 59    angle = sector / vou_span
 60
 61    a = [0, -length / 2, rise - (thickness / 2)]
 62    d = add_vectors(top, [0, -length / 2, (thickness / 2)])
 63
 64    R = Rotation.from_axis_and_angle([0, 1.0, 0], 0.5 * sector, center)
 65    bottom = transform_points([a, d], R)
 66    brick_pts = []
 67    for i in range(vou_span + 1):
 68        R_angle = Rotation.from_axis_and_angle([0, 1.0, 0], -angle * i, center)
 69        points = transform_points(bottom, R_angle)
 70        brick_pts.append(points)
 71
 72    depth = length / vou_length
 73    grouped_data = [pair[0] + pair[1] for pair in zip(brick_pts, brick_pts[1:])]
 74
 75    meshes = []
 76    for i in range(vou_length):
 77        for l, group in enumerate(grouped_data):  # noqa: E741
 78            is_support = l == 0 or l == (len(grouped_data) - 1)
 79            if l % 2 == 0:
 80                point_l = [group[0], group[1], group[2], group[3]]
 81                point_list = [
 82                    [group[0][0], group[0][1] + (depth * i), group[0][2]],
 83                    [group[1][0], group[1][1] + (depth * i), group[1][2]],
 84                    [group[2][0], group[2][1] + (depth * i), group[2][2]],
 85                    [group[3][0], group[3][1] + (depth * i), group[3][2]],
 86                ]
 87                p_t = translate_points(point_l, [0, depth * (i + 1), 0])
 88                vertices = point_list + p_t
 89                faces = [[0, 1, 3, 2], [0, 4, 5, 1], [4, 6, 7, 5], [6, 2, 3, 7], [1, 5, 7, 3], [2, 6, 4, 0]]
 90                mesh = Mesh.from_vertices_and_faces(vertices, faces)
 91                mesh.attributes["is_support"] = is_support
 92                meshes.append(mesh)
 93            else:
 94                point_l = [group[0], group[1], group[2], group[3]]
 95                points_base = translate_points(point_l, [0, depth / 2, 0])
 96                points_b_t = translate_points(points_base, [0, depth * i, 0])
 97                points_t = translate_points(points_base, [0, depth * (i + 1), 0])
 98                vertices = points_b_t + points_t
 99                if i != vou_length - 1:
100                    faces = [[0, 1, 3, 2], [0, 4, 5, 1], [4, 6, 7, 5], [6, 2, 3, 7], [1, 5, 7, 3], [2, 6, 4, 0]]
101                    mesh = Mesh.from_vertices_and_faces(vertices, faces)
102                    mesh.attributes["is_support"] = is_support
103                    meshes.append(mesh)
104
105    for l, group in enumerate(grouped_data):  # noqa: E741
106        is_support = l == 0 or l == (len(grouped_data) - 1)
107        if l % 2 != 0:
108            point_l = [group[0], group[1], group[2], group[3]]
109            p_t = translate_points(point_l, [0, depth / 2, 0])
110            vertices = point_l + p_t
111            faces = [[0, 1, 3, 2], [0, 4, 5, 1], [4, 6, 7, 5], [6, 2, 3, 7], [1, 5, 7, 3], [2, 6, 4, 0]]
112            mesh = Mesh.from_vertices_and_faces(vertices, faces)
113            mesh.attributes["is_support"] = is_support
114            meshes.append(mesh)
115
116            point_f = translate_points(point_l, [0, length, 0])
117            p_f = translate_points(point_f, [0, -depth / 2, 0])
118            vertices = p_f + point_f
119            faces = [[0, 1, 3, 2], [0, 4, 5, 1], [4, 6, 7, 5], [6, 2, 3, 7], [1, 5, 7, 3], [2, 6, 4, 0]]
120            mesh = Mesh.from_vertices_and_faces(vertices, faces)
121            mesh.attributes["is_support"] = is_support
122            meshes.append(mesh)
123
124    # Find the lowest z-coordinate and move all the block to zero.
125    if not zero_is_centerline_or_lowestpoint:
126        min_z = min([min(mesh.vertex_coordinates(key)[2] for key in mesh.vertices()) for mesh in meshes])
127        for mesh in meshes:
128            mesh.translate([0, 0, -min_z])
129
130    frames = []
131    for i, mesh in enumerate(meshes):
132        frame = Frame(mesh.face_polygon(5).frame.point, mesh.vertex_point(0) - mesh.vertex_point(2), mesh.vertex_point(4) - mesh.vertex_point(2))
133        frames.append(frame)
134
135    return meshes, frames
136

139barrel_vault = from_barrel_vault(span=6000, length=6000, thickness=250, rise=600, vou_span=5, vou_length=5)
