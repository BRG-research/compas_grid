  7from compas import json_load
  8from compas.geometry import Frame
  9from compas.geometry import Line
 10from compas.geometry import Point
 11from compas.geometry import Translation
 12from compas.geometry import Vector
 13from compas.geometry.transformation import Transformation
 14from compas_grid.elements import BeamTProfileElement
 15from compas_grid.elements import BeamVProfileElement  # noqa: F401
 16from compas_grid.elements import BlockElement
 17from compas_grid.elements import CableElement
 18from compas_grid.elements import ColumnSquareElement
 19
 20# =============================================================================
 21# JSON file with the geometry of the model.
 22# =============================================================================
 23rhino_geometry = json_load(Path("data/frame.json"))
 24lines = rhino_geometry["lines"]
 25barrel_vault = json_load(Path("data/barrel_vault.json"))
 26
 27# =============================================================================
 28# Model
 29# =============================================================================
 30model = Model()
 31
 32# =============================================================================
 33# Add Elements
 34# =============================================================================
 35
 36# Add columns
 37for i in range(0, 4):
 38    column = ColumnSquareElement(300, 300, lines[i].length)
 39    column.transformation = Transformation.from_frame_to_frame(Frame.worldXY(), Frame(lines[i].start))
 40    model.add_element(column)
 41
 42# Add beams
 43beams = []
 44for i in range(4, len(lines) - 2):
 45    beam = BeamTProfileElement(width=300, height=700, step_width_left=75, step_height_left=150, length=lines[i].length)
 46    point = lines[i].start
 47    xaxis = Vector.Zaxis().cross(lines[i].vector)
 48    yaxis = Vector.Zaxis()
 49    target_frame = Frame(point, xaxis, yaxis)
 50    X = Transformation.from_frame_to_frame(Frame.worldXY(), target_frame)
 51    T = Translation.from_vector([0, beam.height * 0.5, 0])
 52    beam.transformation = X * T
 53    beam.extend(150)
 54    model.add_element(beam)
 55    beams.append(beam)
 56
 57# Add cables
 58cables = []
 59for i in range(6, len(lines)):
 60    cable = CableElement(length=lines[i].length, radius=20)
 61    point = lines[i].start
 62    xaxis = Vector.Zaxis().cross(lines[i].vector)
 63    yaxis = Vector.Zaxis()
 64    target_frame = Frame(point, xaxis, yaxis)
 65    X = Transformation.from_frame_to_frame(Frame.worldXY(), target_frame)
 66    T = Translation.from_vector([0, beam.height * 0.1, 0])
 67    cable.transformation = X * T
 68    cable.extend(200)
 69    model.add_element(cable)
 70    cables.append(cable)
 71
 72# Add barrel vault blocks
 73blocks = []
 74for i in range(len(barrel_vault["meshes"])):
 75    mesh = barrel_vault["meshes"][i]
 76    frame = barrel_vault["frames"][i]
 77    mesh.transform(Transformation.from_frame_to_frame(frame, Frame.worldXY()))
 78    block = BlockElement(shape=mesh, is_support=mesh.attributes["is_support"])
 79    block.transformation = Translation.from_vector([0, 0, 3800]) * Transformation.from_frame_to_frame(Frame.worldXY(), frame)
 80    model.add_element(block)
 81    blocks.append(block)
 82
 83# =============================================================================
 84# Add Interactions
 85# =============================================================================
 86for beam in beams:
 87    for block in blocks:
 88        model.add_interaction(beam, block)
 89        model.add_modifier(beam, block)  # beam -> cuts -> block
 90
 91# =============================================================================
 92# Compute Contacts
 93# =============================================================================
 94model.compute_contacts(tolerance=1, minimum_area=1, k=8)
 95
 96# =============================================================================
 97# Visualize
 98# =============================================================================
 99config = Config()
100config.camera.target = [0, 0, 100]
101config.camera.position = [10000, -10000, 10000]
102config.camera.near = 10
103config.camera.far = 100000
104viewer = Viewer(config=config)
105viewer = Viewer()
106
107lines = []
108for element in list(model.elements()):
109    for line in element.modelgeometry.to_lines():
110        lines.append(Line(Point(*line[0]), Point(*line[1])))
111viewer.scene.add(lines)
112
113for edge in model.graph.edges():
114    if model.graph.edge_attribute(edge, "contacts"):
115        for contact in model.graph.edge_attribute(edge, "contacts"):
116            viewer.scene.add(contact.mesh, facecolor=(0, 255, 0), hide_coplanaredges=True)
117
118viewer.show()
